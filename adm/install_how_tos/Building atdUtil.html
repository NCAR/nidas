Building atdUtil for ADS3

Building for ARM

cd /tmp
cvs -d /code/cvs co isa/src
cd isa
mkdir -p build/arm
cd build/arm
export CFLAGS="-O2 -march=armv5"
export CXXFLAGS="-O2 -march=armv5"
../../src/configure --host=arm-linux --prefix=/net/opt_lnx/ads3/arm
make
make install
make clean

Building for X86

cd /tmp/isa
mkdir build/x86
cd build/x86
export CFLAGS="-O2"
export CXXFLAGS="-O2"
../../src/configure --prefix=/net/opt_lnx/ads3/x86
make
make install
make clean
